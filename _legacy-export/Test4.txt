        function setupSwipeToClose(drawerId, closeFunc) {
            const drawer = document.getElementById(drawerId);
            if (!drawer) return;
            const content = drawer.querySelector('.sheet-content');
            if(!content) return;

            let startY = 0, currentY = 0, isDragging = false;

            content.addEventListener('touchstart', (e) => {
                const scrollArea = content.querySelector('.overflow-y-auto');
                if (scrollArea && scrollArea.contains(e.target) && scrollArea.scrollTop > 0) return;
                
                startY = e.touches[0].clientY;
                isDragging = true;
                content.style.transition = 'none';
            }, { passive: true });

            content.addEventListener('touchmove', (e) => {
                if (!isDragging) return;
                currentY = e.touches[0].clientY;
                const deltaY = currentY - startY;
                if (deltaY > 0) {
                    content.style.transform = `translateY(${deltaY}px)`;
                }
            }, { passive: true });

            content.addEventListener('touchend', (e) => {
                if (!isDragging) return;
                isDragging = false;
                content.style.transition = 'transform 0.3s cubic-bezier(0.2, 0.8, 0.2, 1)';
                const deltaY = currentY - startY;
                
                if (deltaY > 80) {
                    haptic('medium');
                    closeFunc();
                    setTimeout(() => content.style.transform = '', 300);
                } else {
                    content.style.transform = ''; 
                }
            });
        }

function serverCall(functionName, args = []) {
    // Vérification immédiate de la connexion du navigateur
    if (!navigator.onLine) {
        return Promise.reject(new Error("OFFLINE"));
    }

    return new Promise((resolve, reject) => {
        google.script.run
            .withSuccessHandler(res => (res && res.error) ? reject(new Error(res.error)) : resolve(res))
            .withFailureHandler(reject)
            [functionName](...args);
    });
}

        function calculateCurrentWeek() {
            const d = new Date(state.currentDate);
            const dayOfWeek = d.getDay();
            const distanceToMonday = d.getDate() - dayOfWeek + (dayOfWeek === 0 ? -6 : 1);
            
            state.currentWeekKeys = [];
            for (let i = 0; i < 6; i++) { 
                const temp = new Date(d);
                temp.setDate(distanceToMonday + i);
                const year = temp.getFullYear();
                const month = String(temp.getMonth() + 1).padStart(2, '0');
                const day = String(temp.getDate()).padStart(2, '0');
                state.currentWeekKeys.push(`${year}-${month}-${day}`);
            }
            
            const startStr = state.currentWeekKeys[0].split('-').reverse().join('/');
            const endStr = state.currentWeekKeys[5].split('-').reverse().join('/');
            
            document.querySelectorAll('.week-display-dates').forEach(el => el.textContent = `${startStr} au ${endStr}`);
            
            const d1 = new Date(state.currentWeekKeys[0]);
            d1.setHours(0, 0, 0, 0);
            d1.setDate(d1.getDate() + 3 - (d1.getDay() + 6) % 7);
            const week1 = new Date(d1.getFullYear(), 0, 4);
            state.currentWeekNumber = 1 + Math.round(((d1.getTime() - week1.getTime()) / 86400000 - 3 + (week1.getDay() + 6) % 7) / 7);
            
            document.querySelectorAll('.week-display-title').forEach(el => {
                el.innerHTML = `Semaine ${state.currentWeekNumber} <i class="fas fa-chevron-down text-[#00b5e2] text-[9px] ml-1"></i>`;
            });
        }

            for(let w = 1; w <= 52; w++) {

                    start: d.toLocaleDateString('fr-FR', {day:'2-digit', month:'2-digit'}),
                    end: dEnd.toLocaleDateString('fr-FR', {day:'2-digit', month:'2-digit'}),
                    timestamp: d.getTime(),

                });

async function fetchTeamData() {
    let filterValue;
    if (state.teamFilter.length === 0 || state.teamFilter.includes('Tous')) filterValue = 'Tous';
    else if (state.teamFilter.length === 1) filterValue = state.teamFilter[0]; 
    else filterValue = state.teamFilter; 

    const monthStr = state.currentWeekKeys[0].slice(0, 7);
    const offlineIndicator = document.getElementById('offline-indicator');

    try {
        // On tente d'appeler le serveur
        const result = await serverCall('getTeamPresencesForMonth', ['production', filterValue, monthStr, null]);
        state.personnel = result.personnel || [];
        state.teamPresences = result.presences || {};

        // TOUT VA BIEN : On met en cache (sauvegarde locale)
        localStorage.setItem('cachedPersonnel', JSON.stringify(state.personnel));
        localStorage.setItem('cachedPresences', JSON.stringify(state.teamPresences));
        
        // On cache le bandeau hors-ligne s'il était affiché
        offlineIndicator.classList.add('hidden');

    } catch (error) {
        // ERREUR OU HORS-LIGNE : On tente de lire le cache
        if (error.message === "OFFLINE" || !navigator.onLine || error.message.includes("network")) {
            console.log("Passage en mode hors-ligne, lecture du cache...");
            
            const cachedPers = localStorage.getItem('cachedPersonnel');
            const cachedPres = localStorage.getItem('cachedPresences');

            if (cachedPers && cachedPres) {
                state.personnel = JSON.parse(cachedPers);
                state.teamPresences = JSON.parse(cachedPres);
                
                // On affiche le bandeau hors-ligne
                offlineIndicator.classList.remove('hidden');
                showMobileToast("Vous êtes hors-ligne. Affichage des dernières données.", true);
            } else {
                throw new Error("Pas de réseau et aucune donnée en cache.");
            }
        } else {
            // Si c'est une autre erreur (ex: bug serveur), on la remonte normalement
            throw error;
        }
    }
}

        function handleTeamFilterChange(cb) {
            haptic('light');
            const val = cb.value;

// --- VUE 1 : ÉQUIPE (RENDU CARTES, TRI & SÉPARATEURS) ---
        function renderMobileCards() {
            const container = document.getElementById('mobile-team-list');
            const daysNames = ['Lu', 'Ma', 'Me', 'Je', 'Ve', 'Sa']; 
            const todayStr = new Date().toISOString().slice(0, 10);
            const searchQuery = document.getElementById('mobile-search-input').value.toLowerCase().trim();

            // 1. Filtrage habituel
            let filteredPersonnel = state.personnel.filter(p => {
                const fullName = `${p.prenom} ${p.nom}`.toLowerCase();
                const matchSearch = fullName.includes(searchQuery);
                let matchShift = true;
                if (state.shiftFilter !== 'Tous') {
                    const mondayStr = state.currentWeekKeys[0];
                    const actualShift = (state.teamPresences[p.id] && state.teamPresences[p.id][mondayStr]) ? state.teamPresences[p.id][mondayStr].s : null;
                    const theoreticalShift = p.quartDefaut;
                    matchShift = (actualShift === state.shiftFilter || (!actualShift && theoreticalShift === state.shiftFilter));
                }
                return matchSearch && matchShift;
            });

            if (filteredPersonnel.length === 0) {
                container.innerHTML = `<div class="bg-white p-8 rounded-[2rem] text-center text-gray-400 font-bold shadow-sm">Aucun collaborateur trouvé.</div>`;
                return;
            }

            // 2. TRI : Par Poste de travail d'abord, puis par Ordre Alphabétique (Nom)
            filteredPersonnel.sort((a, b) => {
                // "zzz" permet de repousser ceux qui n'ont pas de poste à la toute fin de la liste
                const posteA = (a.posteDeTravail || 'zzz').toLowerCase();
                const posteB = (b.posteDeTravail || 'zzz').toLowerCase();
                
                if (posteA < posteB) return -1;
                if (posteA > posteB) return 1;
                
                // Si c'est le même poste, on trie par nom de famille
                const nomA = (a.nom || '').toLowerCase();
                const nomB = (b.nom || '').toLowerCase();
                if (nomA < nomB) return -1;
                if (nomA > nomB) return 1;
                return 0;
            });

            let fullHtmlContent = '';
            let currentPoste = null; // Mémorise le poste en cours d'affichage

            // 3. Rendu des cartes et des séparateurs
            filteredPersonnel.forEach(person => {
                
                const displayPoste = person.posteDeTravail || 'Non assigné';
                
                // SÉPARATEUR VISUEL : Si le poste change, on affiche un en-tête
                if (currentPoste !== displayPoste) {
                    // Marge réduite pour le premier élément, plus grande pour les suivants
                    const marginClass = currentPoste === null ? "mb-3" : "mt-6 mb-3";
                    
                    fullHtmlContent += `
                        <div class="flex items-center gap-3 ${marginClass} mx-1">
                            <span class="text-[10px] font-black text-[#00205b] uppercase tracking-widest bg-gray-50 px-3 py-1.5 rounded-lg border border-gray-200/60 shadow-sm">${displayPoste}</span>
                            <div class="h-[2px] flex-grow bg-gradient-to-r from-gray-200/80 to-transparent rounded-full"></div>
                        </div>
                    `;
                    currentPoste = displayPoste; // On met à jour le poste actuel
                }

                let daysHtml = '';
                let commentsHtml = ''; 
                
                state.currentWeekKeys.forEach((dateKey, index) => {
                    const dayData = (state.teamPresences[person.id] && state.teamPresences[person.id][dateKey]) || {};
                    const status = dayData.s || '';
                    const comment = dayData.c || '';
                    const hs = dayData.hs || 0;
                    const isToday = dateKey === todayStr;
                    
                    if (comment) {
                        commentsHtml += `
                            <div class="flex items-start gap-2.5 bg-blue-50/50 p-2.5 rounded-xl border border-blue-100/50">
                                <div class="mt-[2px] shrink-0 w-4 flex justify-center"><i class="fas fa-comment-dots text-[#00b5e2] text-[11px]"></i></div>
                                <div class="leading-snug">
                                    <span class="text-[9px] font-black text-[#00205b] uppercase tracking-widest mr-1">${daysNames[index]}:</span>
                                    <span class="text-[11px] font-bold text-[#00205b]">${comment}</span>
                                </div>
                            </div>
                        `;
                    }
                    
                    let bgClass = 'bg-gray-100 text-gray-400 border border-gray-200';
                    let label = '-';
                    
                    if (status === 'M') { bgClass = 'bg-[#A5DFF0] text-[#21758A] ring-2 ring-[#A5DFF0] ring-offset-1'; label = 'M'; }
                    else if (status === 'A') { bgClass = 'bg-blue-100 text-blue-700 ring-2 ring-blue-400 ring-offset-1'; label = 'A'; }
                    else if (status === 'N') { bgClass = 'bg-[#9cb4d4] text-[#00205b] ring-2 ring-[#9cb4d4] ring-offset-1'; label = 'N'; }
                    else if (['Abs', 'Ma', 'LMa', 'RF'].includes(status)) { bgClass = 'bg-red-100 text-red-600 ring-2 ring-red-400 ring-offset-1'; label = status; }
                    else if (['CP', '4HCP', 'JRTT'].includes(status)) { bgClass = 'bg-green-100 text-green-700 ring-2 ring-green-400 ring-offset-1'; label = status; }
                    else if (status) { bgClass = 'bg-orange-100 text-orange-700'; label = status; }

                    const todayIndicator = isToday ? `<div class="w-1.5 h-1.5 bg-[#00b5e2] rounded-full absolute -top-1 shadow-sm z-20"></div>` : '';
                    const commentIndicator = comment ? `<div class="w-2 h-2 bg-[#00b5e2] border border-white rounded-full absolute top-0 right-0 z-10"></div>` : '';
                    const hsIndicator = hs !== 0 ? `<div class="absolute -bottom-1.5 -left-1.5 ${hs > 0 ? 'bg-[#00205b]' : 'bg-red-500'} text-white text-[7px] font-black px-1 rounded border border-white z-10">${hs > 0 ? '+' : ''}${hs}h</div>` : '';
                    const todayTextClass = isToday ? 'text-[#00b5e2]' : 'text-gray-400';

                    daysHtml += `
                        <div class="flex flex-col items-center gap-0.5 cursor-pointer relative card-pop" onclick="openQuickAction('${person.id}', '${dateKey}')">
                            ${todayIndicator}
                            <span class="text-[8px] font-black uppercase ${todayTextClass}">${daysNames[index]}</span>
                            <div class="w-9 h-9 sm:w-10 sm:h-10 rounded-xl flex items-center justify-center font-black text-[11px] sm:text-xs transition-transform active:scale-90 relative ${bgClass}">
                                ${label}
                                ${commentIndicator}
                                ${hsIndicator}
                            </div>
                        </div>
                    `;
                });

                const initial = person.role === 'Compagnon' ? 'C' : (person.role === 'Intérimaire' ? 'I' : person.role.charAt(0));
                const initialColor = person.role === 'Compagnon' ? 'bg-[#175bca] text-white' : 'bg-gray-200 text-gray-600';

// On ajoute le wrapper de swipe dynamique
fullHtmlContent += `
    <div class="swipe-wrapper" data-person-id="${person.id}">
        <div class="swipe-bg">
            <div class="swipe-icon-left"><i class="fas fa-check"></i></div>
            <div class="swipe-icon-right"><i class="fas fa-times"></i></div>
        </div>
        
        <div class="swipe-front p-3.5 rounded-[2rem] shadow-sm border border-gray-100">
            <div class="flex items-center justify-between mb-3">
                <div class="flex items-center gap-3 cursor-pointer card-pop w-full" onclick="openYearlySchedule('${person.id}')">
                    <div class="w-10 h-10 rounded-2xl flex items-center justify-center font-black text-sm ${initialColor} shrink-0">${initial}</div>
                    <div class="truncate">
                        <h2 class="font-black text-[15px] leading-none text-[#00205b] truncate pr-2">${person.nom} <span class="font-normal text-gray-500">${person.prenom}</span></h2>
                        <p class="text-[9px] font-bold text-[#00b5e2] uppercase tracking-widest mt-1 truncate">${person.posteDeTravail || 'Non assigné'}</p>
                    </div>
                    <div class="ml-auto w-7 h-7 rounded-full bg-gray-50 flex items-center justify-center text-[#00b5e2] shrink-0 text-xs"><i class="fas fa-calendar-alt"></i></div>
                </div>
            </div>
            <div class="flex justify-between items-center bg-gray-50 p-2 px-3 rounded-[1.25rem] border border-gray-100">
                ${daysHtml}
            </div>
            ${commentsHtml ? `<div class="mt-2.5 flex flex-col gap-1.5">${commentsHtml}</div>` : ''}
        </div>
    </div>
`;
            });
container.innerHTML = fullHtmlContent;
    initSwipeActions(); // <-- AJOUTER CETTE LIGNE ICI
}

        async function renderYearlySchedule() {
            const year = state.yearlyCalendarDate.getFullYear();
            
            // 1. Affichage immédiat du calendrier à vide (Création de la structure visuelle)
            buildYearlyMonthGrid();
            
            if (!state.yearlyPresences[state.yearlyPersonId] || !state.yearlyPresences[state.yearlyPersonId][`${year}-01-01`]) {
                // 2. Remplacement du gros loader bloquant par un mini spinner discret dans les détails
                document.getElementById('ys-month-details').innerHTML = `
                    <div class="flex flex-col items-center justify-center py-6 gap-2">
                        <div class="w-6 h-6 border-2 border-[#00b5e2]/30 border-t-[#00b5e2] rounded-full animate-spin"></div>
                        <span class="text-[9px] font-bold text-gray-400 uppercase tracking-widest">Récupération des données...</span>
                    </div>
                `;
                
                try {
                    // Requête silencieuse en arrière-plan
                    const data = await serverCall('getPresencesForYear', [state.yearlyPersonId, year]);
                    if(!state.yearlyPresences[state.yearlyPersonId]) state.yearlyPresences[state.yearlyPersonId] = {};
                    Object.assign(state.yearlyPresences[state.yearlyPersonId], data);
                    
                    // 3. Dès réception, on "colorie" la grille sans que l'utilisateur ne l'ait vue bugger
                    buildYearlyMonthGrid();
                } catch(e) {
                    showMobileToast("Erreur de chargement", true);
                    document.getElementById('ys-month-details').innerHTML = '<p class="text-[10px] text-red-400 font-bold uppercase tracking-widest text-center mt-4">Erreur de chargement</p>';
                }
            }
        }

        function buildYearlyMonthGrid() {
            const grid = document.getElementById('ys-calendar-grid');
            const detailsContainer = document.getElementById('ys-month-details');
            grid.innerHTML = '';
            detailsContainer.innerHTML = '';
            
            const year = state.yearlyCalendarDate.getFullYear();
            const month = state.yearlyCalendarDate.getMonth();
            const monthsNames = ['Janvier', 'Février', 'Mars', 'Avril', 'Mai', 'Juin', 'Juillet', 'Août', 'Septembre', 'Octobre', 'Novembre', 'Décembre'];
            
            document.getElementById('ys-month-title').textContent = `${monthsNames[month]} ${year}`;

            const firstDay = new Date(year, month, 1).getDay();
            const daysInMonth = new Date(year, month + 1, 0).getDate();
            const emptySlots = firstDay === 0 ? 6 : firstDay - 1;

            let workDaysCount = 0, absCount = 0;
            let gridHtml = '';
            let monthDetailsHtml = '';

            for (let i = 0; i < emptySlots; i++) gridHtml += `<div class="aspect-square"></div>`;

            const todayStr = new Date().toISOString().slice(0, 10);
            const myData = state.yearlyPresences[state.yearlyPersonId] || {};

            for (let i = 1; i <= daysInMonth; i++) {
                const dateKey = `${year}-${String(month + 1).padStart(2, '0')}-${String(i).padStart(2, '0')}`;
                const dayData = myData[dateKey] || {};
                const status = dayData.s || '';
                const comment = dayData.c || '';
                const hs = dayData.hs || 0;
                const isToday = dateKey === todayStr;

                if (['M', 'A', 'N', 'J', 'SC', 'SL'].includes(status)) workDaysCount++;
                if (['Abs', 'CP', 'Ma', 'LMa', 'RF'].includes(status)) absCount++;

                let bgClass = 'bg-gray-50 text-[#00205b] border border-gray-100';
                if (status === 'M') bgClass = 'bg-emerald-100 text-emerald-700 border-emerald-200';
                else if (status === 'A') bgClass = 'bg-blue-100 text-blue-700 border-blue-200';
                else if (status === 'N') bgClass = 'bg-indigo-100 text-indigo-700 border-indigo-200';
                else if (['Abs', 'Ma', 'LMa', 'RF'].includes(status)) bgClass = 'bg-red-100 text-red-600 border-red-200';
                else if (['CP', '4HCP', 'JRTT'].includes(status)) bgClass = 'bg-green-100 text-green-700 border-green-200';
                else if (status) bgClass = 'bg-orange-100 text-orange-700 border-orange-200';

                const todayStyle = isToday ? 'ring-2 ring-[#00b5e2] ring-offset-1 font-black' : 'font-bold';
                const commentIndicator = comment ? `<div class="w-1.5 h-1.5 bg-[#00b5e2] rounded-full absolute top-1 right-1"></div>` : '';
                const hsIndicator = hs !== 0 ? `<div class="w-1.5 h-1.5 ${hs > 0 ? 'bg-[#00205b]' : 'bg-red-500'} rounded-full absolute bottom-1 right-1"></div>` : '';

                gridHtml += `
                    <div onclick="openQuickAction('${state.yearlyPersonId}', '${dateKey}', true)" class="cursor-pointer aspect-square rounded-xl flex flex-col items-center justify-center relative ${bgClass} ${todayStyle} active:scale-95 transition-transform card-pop">
                        <span class="text-[12px]">${i}</span>
                        ${status ? `<span class="text-[8px] uppercase tracking-tighter mt-0.5 opacity-80 truncate w-full text-center px-1">${status}</span>` : ''}
                        ${commentIndicator}
                        ${hsIndicator}
                    </div>
                `;

                if (comment || hs !== 0) {
                    const dateObj = new Date(dateKey);
                    const dateFormatted = dateObj.toLocaleDateString('fr-FR', { weekday: 'short', day: 'numeric', month: 'short' });
                    monthDetailsHtml += `
                        <div class="bg-white p-3 rounded-2xl border border-gray-100 shadow-sm flex items-start gap-3">
                            <div class="w-10 h-10 rounded-xl bg-[#00205b]/5 flex flex-col items-center justify-center shrink-0">
                                <span class="text-[8px] font-black text-gray-400 uppercase tracking-widest">${dateFormatted.split(' ')[0]}</span>
                                <span class="text-sm font-black text-[#00205b] leading-none">${i}</span>
                            </div>
                            <div class="flex-grow pt-0.5">
                                ${hs !== 0 ? `<div class="inline-block px-2 py-0.5 rounded-md text-[9px] font-black uppercase mb-1 ${hs > 0 ? 'bg-indigo-100 text-[#00205b]' : 'bg-red-100 text-red-600'}">HS: ${hs > 0 ? '+' : ''}${hs}h</div>` : ''}
                                ${comment ? `<p class="text-xs font-bold text-[#00205b]"><i class="fas fa-comment-dots text-[#00b5e2] text-[10px] mr-1"></i> ${comment}</p>` : ''}
                            </div>
                        </div>
                    `;
                }

function renderViewStats() {
    if (!state.indicatorData || state.indicatorData.error) return;
    const data = state.indicatorData;

    // 1. CALCUL DES COMPTEURS DU HAUT
    const dC = data.compagnons?.weekly || {};
    const dI = data.interimaires?.weekly || {};
    const activityStatusCodes = ['presence', 'F', '4HF', 'D', 'P', 'S'];
    
    const totalPresent = activityStatusCodes.reduce((sum, code) => sum + (dC[code]?.length || 0) + (dI[code]?.length || 0), 0);
    const totalAbsences = ['Ma', 'LMa', 'CP', '4HCP', 'Abs', 'RF', 'JRTT'].reduce((sum, key) => sum + (dC[key]?.length || 0) + (dI[key]?.length || 0), 0);
    
    const workforceTotal = data.workforceTotals?.total || 0;
    const theoreticalTotal = workforceTotal * 5; // Base 5 jours ouvrés
    const presenceRate = theoreticalTotal > 0 ? Math.round((totalPresent / theoreticalTotal) * 100) : 0;

    // Mise à jour de l'affichage des compteurs
    document.getElementById('stat-total-staff').textContent = workforceTotal;
    document.getElementById('stat-total-abs').textContent = totalAbsences;
    document.getElementById('stat-presence-rate').textContent = `${presenceRate}%`;
    
    const barEl = document.getElementById('stat-presence-bar');
    if (barEl) {
        barEl.style.width = `${presenceRate}%`;
        if (presenceRate < 85) barEl.className = 'h-full bg-red-500 transition-all duration-1000';
        else if (presenceRate < 95) barEl.className = 'h-full bg-orange-400 transition-all duration-1000';
        else barEl.className = 'h-full bg-emerald-500 transition-all duration-1000';
    }

    // 2. GÉNÉRATION DU GRAPHIQUE PARETO
    const breakdown = data.monthlyAbsenceBreakdown || {};
    const sortedAbsences = Object.entries(breakdown).sort(([, a], [, b]) => b - a);
    const totalAbsPareto = sortedAbsences.reduce((sum, item) => sum + item[1], 0);
    
    let cumulativeCount = 0;
    const paretoLabels = [];
    const paretoCounts = [];
    const paretoLine = [];

    sortedAbsences.forEach(([type, count]) => {
        cumulativeCount += count;
        paretoLabels.push(type);
        paretoCounts.push(count);
        paretoLine.push(totalAbsPareto > 0 ? (cumulativeCount / totalAbsPareto) * 100 : 0);
    });

    const ctx = document.getElementById('mobile-pareto-chart').getContext('2d');
    if (mobileParetoChartInstance) mobileParetoChartInstance.destroy();
    
    mobileParetoChartInstance = new Chart(ctx, {
        type: 'bar',
        data: {
            labels: paretoLabels,
            datasets: [{
                label: "Nombre",
                data: paretoCounts,
                backgroundColor: '#00205b',
                borderRadius: 6,
                barPercentage: 0.6
            }, {
                type: 'line',
                label: '% Cumulé',
                data: paretoLine,
                borderColor: '#00b5e2',
                borderWidth: 2,
                pointBackgroundColor: '#00b5e2',
                yAxisID: 'y1',
                datalabels: { display: false }
            }]
        },
        options: {
            responsive: true, maintainAspectRatio: false,
            scales: {
                x: { grid: { display: false }, ticks: { font: { size: 10, weight: 'bold' } } },
                y: { display: false },
                y1: { display: false, min: 0, max: 100 }
            },
            plugins: {
                legend: { display: false },
                datalabels: { 
                    anchor: 'end', align: 'top', color: '#00205b', 
                    font: { weight: 'black', size: 10 },
                    formatter: Math.round 
                }
            }
        }
    });

// 3. TABLEAU DES ABSENCES INTERACTIF
    renderAbsencesList('week');

    // 4. BARRES DE PROGRESSION (Effectifs par Section/Poste)
    const renderProgressBar = (present, total, label) => {
        const percentage = total > 0 ? (present / total) * 100 : 0;
        const color = (present >= total && total > 0) ? 'bg-emerald-500' : (present >= total - 1 && total > 0) ? 'bg-orange-500' : 'bg-red-500';
        const textCol = (present >= total && total > 0) ? 'text-emerald-600' : (present >= total - 1 && total > 0) ? 'text-orange-600' : 'text-red-500';
        
        return `
            <div class="mb-3">
                <div class="flex justify-between text-[10px] font-black uppercase tracking-widest mb-1">
                    <span class="text-[#00205b] truncate pr-2">${label}</span>
                    <span class="${textCol}">${present}/${total}</span>
                </div>
                <div class="w-full bg-gray-100 rounded-full h-1.5 overflow-hidden">
                    <div class="${color} h-full rounded-full transition-all duration-500" style="width: ${Math.min(percentage, 100)}%"></div>
                </div>
            </div>`;
    };

    let secHtml = '';
    if(data.sectionHeadcount?.theoretical) {
        Object.keys(data.sectionHeadcount.theoretical).sort().forEach(sec => { 
            secHtml += renderProgressBar(data.sectionHeadcount.presentToday[sec] || 0, data.sectionHeadcount.theoretical[sec] || 0, `Sec. ${sec}`); 
        });
    }
    document.getElementById('stat-sections-container').innerHTML = secHtml || '<p class="text-xs text-gray-400">Aucune donnée section</p>';

    let wsHtml = '';
    if(data.workstationHeadcount?.theoretical) {
        Object.keys(data.workstationHeadcount.theoretical).sort().forEach(ws => { 
            wsHtml += renderProgressBar(data.workstationHeadcount.presentToday[ws] || 0, data.workstationHeadcount.theoretical[ws] || 0, ws); 
        });
    }
    document.getElementById('stat-postes-container').innerHTML = wsHtml || '<p class="text-xs text-gray-400">Aucune donnée poste</p>';
}

        function renderAbsencesList(daySelector) {
            if (window.navigator && window.navigator.vibrate) haptic('light');

            document.querySelectorAll('.abs-day-btn').forEach(btn => {

                    btn.className = "abs-day-btn flex-1 py-1.5 rounded-lg text-[10px] font-black uppercase tracking-widest bg-[#00205b] text-white shadow-sm card-pop";
                } else {
                    btn.className = "abs-day-btn flex-1 py-1.5 rounded-lg text-[10px] font-bold uppercase tracking-widest text-gray-400 bg-white border border-gray-100 card-pop";
                }
            });

            const absContainer = document.getElementById('absences-interactive-list');
            if(!state.currentWeekKeys || state.currentWeekKeys.length === 0) return;

            // 2. Déterminer quelle(s) date(s) on analyse
            let datesToCheck = [];
            if (daySelector === 'week') {
                datesToCheck = state.currentWeekKeys.slice(0, 5); // Lundi à Vendredi
            } else {
                datesToCheck = [state.currentWeekKeys[parseInt(daySelector)]];
            }

            const absenceTypes = ['Abs', 'CP', '4HCP', 'Ma', 'LMa', 'RF', 'JRTT']; 
            let groupedAbsences = {};

            // 3. On fouille dans l'application pour trouver les noms
            state.personnel.forEach(person => {
                datesToCheck.forEach(dateKey => {
                    const dayData = (state.teamPresences[person.id] && state.teamPresences[person.id][dateKey]) || {};
                    const status = dayData.s;
                    
                    if (status && absenceTypes.includes(status)) {
                        if (!groupedAbsences[status]) groupedAbsences[status] = new Set();
                        // L'utilisation d'un "Set" empêche d'afficher un nom en double si la personne est absente toute la semaine
                        groupedAbsences[status].add(`${person.nom} ${person.prenom}`);
                    }
                });
            });

            // 4. On génère l'affichage HTML
            let absHtml = '';
            const sortedTypes = Object.keys(groupedAbsences).sort((a, b) => groupedAbsences[b].size - groupedAbsences[a].size);

            if (sortedTypes.length === 0) {
                absHtml = '<p class="text-center py-4 text-xs text-gray-400 italic font-bold">Aucune absence sur cette sélection.</p>';
            } else {
                sortedTypes.forEach(type => {
                    const count = groupedAbsences[type].size;
                    const label = STATUSES[type] || type;
                    const names = Array.from(groupedAbsences[type]).join(' • '); // Sépare les noms par un point visuel
                    
                    absHtml += `
                        <div class="border border-gray-100 rounded-2xl overflow-hidden mb-2">
                            <button onclick="this.nextElementSibling.classList.toggle('hidden'); if(window.navigator.vibrate) navigator.vibrate(10);" 
                                    class="w-full flex items-center justify-between p-4 bg-gray-50/50 active:bg-gray-100 transition-colors outline-none">
                                <div class="flex items-center gap-3">
                                    <span class="w-8 h-8 rounded-lg bg-white shadow-sm flex items-center justify-center text-[10px] font-black text-[#00205b] border border-gray-100">${type}</span>
                                    <span class="text-xs font-black text-[#00205b] uppercase tracking-tight">${label}</span>
                                </div>
                                <div class="flex items-center gap-2">
                                    <span class="bg-[#00205b] text-white text-[10px] font-black px-2 py-0.5 rounded-full">${count}</span>
                                    <i class="fas fa-chevron-down text-[10px] text-gray-400"></i>
                                </div>
                            </button>
                            <div class="hidden p-4 bg-white border-t border-gray-50">
                                <p class="text-[11px] font-bold text-gray-600 leading-relaxed pr-2">
                                    <i class="fas fa-user text-[#00b5e2] mr-2"></i> ${names}
                                </p>
                            </div>
                        </div>
                    `;
                });
            }
            absContainer.innerHTML = absHtml;
        }

        async function saveAllMobilePresenceDetails(forceStatus = null) {
            haptic('success');
            
            if (forceStatus !== null) state.qaCurrentStatus = forceStatus;
            closeQuickAction();
            
            const pId = state.selectedPersonnelId;
            const dKey = state.selectedDateKey;
            
            const newStatus = state.qaCurrentStatus;
            const newHs = parseFloat(document.getElementById('qa-hs').value) || 0;
            const newComment = document.getElementById('qa-comment').value.trim();

            const isYearly = !document.getElementById('yearly-sheet').classList.contains('hidden');
            const targetDict = isYearly ? state.yearlyPresences : state.teamPresences;

            const oldData = (targetDict[pId] && targetDict[pId][dKey]) ? {...targetDict[pId][dKey]} : {};

            // MAJ optimiste
            if (!targetDict[pId]) targetDict[pId] = {};
            if (!targetDict[pId][dKey]) targetDict[pId][dKey] = {};
            
            const target = targetDict[pId][dKey];
            if (newStatus === '') delete target.s; else target.s = newStatus;
            if (newHs === 0) delete target.hs; else target.hs = newHs;
            if (newComment === '') delete target.c; else target.c = newComment;

            if (Object.keys(target).length === 0) delete targetDict[pId][dKey];
            
            if (isYearly) {
                buildYearlyMonthGrid();
                if (state.teamPresences[pId] && state.currentWeekKeys.includes(dKey)) {
                    state.teamPresences[pId][dKey] = {...target};
                    if (Object.keys(target).length === 0) delete state.teamPresences[pId][dKey];
                    renderMobileCards();
                }
            } else {
                renderMobileCards(); 
            }

            try {
                if (oldData.s !== newStatus) {
                    await serverCall('setPresence', [pId, dKey, newStatus]);
                }
                if (oldData.hs !== newHs || oldData.c !== newComment) {
                    await serverCall('savePresenceDetails', [pId, dKey, newComment, newHs]);
                }
                showMobileToast("Enregistré avec succès !");
            } catch (e) {
                showMobileToast("Erreur réseau. Annulation.", true);
                if (Object.keys(oldData).length === 0) delete targetDict[pId][dKey];
                else targetDict[pId][dKey] = oldData;
                
                if (isYearly) buildYearlyMonthGrid(); else renderMobileCards();
            }
        }

        const personId = wrapper.dataset.personId;
        
        let startX = 0;
        let startY = 0; // On ajoute Y pour détecter le scroll
        let currentX = 0;
        let isDragging = false;
        let isScrolling = false; // Différencie le scroll du glissement horizontal

        wrapper.addEventListener('touchstart', (e) => {
            startX = e.touches[0].clientX;
            startY = e.touches[0].clientY;
            currentX = startX; // CORRECTION CRITIQUE ICI : un clic pur donnera diffX = 0
            
            isDragging = true;
            isScrolling = false;
            front.style.transition = 'none';
        }, { passive: true });

        wrapper.addEventListener('touchmove', (e) => {
            if (!isDragging) return;
            
            currentX = e.touches[0].clientX;
            const currentY = e.touches[0].clientY;
            
            const diffX = currentX - startX;
            const diffY = currentY - startY;

            // SÉCURITÉ : Si on bouge plus verticalement qu'horizontalement, c'est un scroll.
            if (!isScrolling && Math.abs(diffY) > Math.abs(diffX)) {
                isScrolling = true;
                isDragging = false; // On annule le swipe
                front.style.transform = 'translateX(0)';
                return;
            }
            
            // Si l'utilisateur est en train de scroller, on bloque l'animation du swipe
            if (isScrolling) return;

            // Animation dynamique
            if (diffX > 0) {
                bg.style.backgroundColor = '#10b981'; // Vert
                const ratio = Math.min(diffX / SWIPE_THRESHOLD, 1);
                iconLeft.style.opacity = ratio;
                iconLeft.style.transform = `scale(${0.5 + (ratio * 0.5)})`;
                iconRight.style.opacity = 0;
            } else if (diffX < 0) {
                bg.style.backgroundColor = '#ef4444'; // Rouge
                const ratio = Math.min(Math.abs(diffX) / SWIPE_THRESHOLD, 1);
                iconRight.style.opacity = ratio;
                iconRight.style.transform = `scale(${0.5 + (ratio * 0.5)})`;
                iconLeft.style.opacity = 0;
            }

            // Mouvement de la carte
            if (diffX > -120 && diffX < 120) {
                front.style.transform = `translateX(${diffX}px)`;
            } else {
                const resistance = diffX > 0 ? 120 + (diffX - 120) * 0.2 : -120 + (diffX + 120) * 0.2;
                front.style.transform = `translateX(${resistance}px)`;
            }
        }, { passive: true });

        wrapper.addEventListener('touchend', (e) => {
            if (!isDragging) return; // Si on scrollait, on ignore la fin du touché
            isDragging = false;
            
            const diffX = currentX - startX;
            front.style.transition = 'transform 0.3s cubic-bezier(0.2, 0.8, 0.2, 1)';
            
            // Un clic normal a un diffX de 0 (ou très proche), donc il tombera dans le 'else' et s'annulera silencieusement
            if (diffX > SWIPE_THRESHOLD) {
                front.style.transform = `translateX(120%)`;
                handleQuickSwipeAction(personId, 'present');
                setTimeout(() => front.style.transform = 'translateX(0)', 300);
            } else if (diffX < -SWIPE_THRESHOLD) {
                front.style.transform = `translateX(-120%)`;
                handleQuickSwipeAction(personId, 'absent');
                setTimeout(() => front.style.transform = 'translateX(0)', 300);
            } else {
                front.style.transform = 'translateX(0)';
            }
        });
    });
}

async function handleQuickSwipeAction(personId, action) {
    haptic('success');
    
    // On cible la date d'aujourd'hui (format local YYYY-MM-DD)
    const today = new Date();
    // On force le fuseau horaire local pour éviter le décalage UTC
    const todayStr = new Date(today.getTime() - (today.getTimezoneOffset() * 60000)).toISOString().slice(0, 10);
    
    // 1. Intelligence du statut via notre nouvelle fonction !
    let newStatus = 'Abs';
    if (action === 'present') {
        newStatus = determineIntelligentShift(personId, todayStr);
    }
    
    // 2. Mise à jour instantanée de l'interface pour la fluidité
    if (!state.teamPresences[personId]) state.teamPresences[personId] = {};
    if (!state.teamPresences[personId][todayStr]) state.teamPresences[personId][todayStr] = {};
    
    state.teamPresences[personId][todayStr].s = newStatus;
    
    // 3. Rendu visuel et notification
    renderMobileCards();
    
    const toastMsg = action === 'present' 
        ? `Présent : Affecté en ${newStatus}` 
        : `Absence déclarée`;
    showMobileToast(toastMsg);

    // 4. Sauvegarde silencieuse côté serveur (Google Apps Script)
    try {
        await serverCall('setPresence', [personId, todayStr, newStatus]);
    } catch (e) {
        showMobileToast("Erreur de sauvegarde", true);
        // Si besoin, on pourrait faire un rollback ici en cas d'échec
    }
}

// Fonction utilitaire pour trouver le statut dominant d'une semaine précise
function getDominantShiftForWeek(personId, dateObj) {
    const d = new Date(dateObj);
    d.setHours(12, 0, 0, 0); // Sécurité contre les décalages horaires
    
    // Trouver le lundi de cette semaine
    const day = d.getDay();
    const diff = d.getDate() - day + (day === 0 ? -6 : 1);
    const monday = new Date(d.setDate(diff));

    const workShifts = ['M', 'A', 'N', 'J', 'SC', 'SL'];
    let counts = {};
    
    // Analyser du Lundi au Samedi
    for(let i = 0; i < 6; i++) {
        const current = new Date(monday);
        current.setDate(monday.getDate() + i);
        const dateStr = current.toISOString().slice(0, 10);
        
        const dayData = (state.teamPresences[personId] && state.teamPresences[personId][dateStr]);
        if (dayData && workShifts.includes(dayData.s)) {
            counts[dayData.s] = (counts[dayData.s] || 0) + 1;
        }
    }
    
    let dominant = null;
    let max = 0;
    for (const [s, c] of Object.entries(counts)) {
        if (c > max) { max = c; dominant = s; }
    }
    return dominant;
}

// Le vrai cerveau qui analyse les rotations
function determineIntelligentShift(personId, targetDateStr) {
    const person = state.personnel.find(p => p.id === personId);
    // On met en majuscules pour éviter les erreurs de casse (ex: "3*8" ou "3X8")
    const typeQuart = (person && person.typeQuart) ? person.typeQuart.toUpperCase() : ''; 
    const targetDate = new Date(targetDateStr);
    targetDate.setHours(12, 0, 0, 0);

    // 1. Y a-t-il déjà un quart de travail défini cette semaine ?
    let currentWeekShift = getDominantShiftForWeek(personId, targetDate);
    if (currentWeekShift) return currentWeekShift;

    // --- On prépare l'analyse des roulements ---
    const is2x8 = typeQuart.includes('2*8') || typeQuart.includes('2X8');
    const is3x8 = typeQuart.includes('3*8') || typeQuart.includes('3X8');

    // 2. Si non (ex: semaine de CP), on regarde la SEMAINE PRÉCÉDENTE
    const prevWeekDate = new Date(targetDate);
    prevWeekDate.setDate(targetDate.getDate() - 7);
    let prevWeekShift = getDominantShiftForWeek(personId, prevWeekDate);

    if (prevWeekShift) {
        if (is2x8) {
            // Roulement 2*8 : Matin -> Aprem / Aprem -> Matin
            if (prevWeekShift === 'M') return 'A';
            if (prevWeekShift === 'A') return 'M';
        }
        if (is3x8) { 
            // Roulement 3*8 (Ton modèle : Matin -> Nuit -> Après-Midi)
            if (prevWeekShift === 'M') return 'N';
            if (prevWeekShift === 'N') return 'A';
            if (prevWeekShift === 'A') return 'M';
        }
        return prevWeekShift; // S'il n'est ni 2*8 ni 3*8, on répète son quart
    }

    // 3. Toujours rien ? (Ex: CP semaine dernière aussi). On regarde la SEMAINE SUIVANTE
    const nextWeekDate = new Date(targetDate);
    nextWeekDate.setDate(targetDate.getDate() + 7);
    let nextWeekShift = getDominantShiftForWeek(personId, nextWeekDate);

    if (nextWeekShift) {
        if (is2x8) {
            // Logique inversée pour remonter le temps
            if (nextWeekShift === 'M') return 'A';
            if (nextWeekShift === 'A') return 'M';
        }
        if (is3x8) { 
            // Logique inversée (M <- N <- A)
            if (nextWeekShift === 'N') return 'M';
            if (nextWeekShift === 'A') return 'N';
            if (nextWeekShift === 'M') return 'A';
        }
        return nextWeekShift;
    }

    // 4. PLAN C : Aucun historique trouvé, on utilise son quart par défaut
    return (person && person.quartDefaut) ? person.quartDefaut : 'J';
}
