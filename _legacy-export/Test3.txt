    if (schedules && schedules.length > 0) { 
        // -------------------------------------------------------------
        // 1. FUSION DE TOUTES LES ÉQUIPES EN UNE SEULE LISTE
        // -------------------------------------------------------------
        var combinedMembers = [];
        var combinedSchedule = {};
        var teamNames = [];
        var weekDates = schedules[0].weekDates;
        var addedIds = {}; // Pour éviter les doublons

        for (var i = 0; i < schedules.length; i++) {
            var data = schedules[i];
            
            // Collecter les noms d'équipes pour le grand titre
            if (data.teamName && data.teamName !== "Tous" && teamNames.indexOf(data.teamName) === -1) {
                teamNames.push(data.teamName);
            }
            
            // Fusionner le personnel
            if (data.teamMembers) {
                for (var j = 0; j < data.teamMembers.length; j++) {
                    var member = data.teamMembers[j];
                    if (!addedIds[member.id]) {
                        combinedMembers.push(member);
                        addedIds[member.id] = true;
                    }
                }
            }
            
            // Fusionner les plannings
            if (data.schedule) {
                for (var personId in data.schedule) {
                    combinedSchedule[personId] = data.schedule[personId];
                }
            }
        }

        // Titre dynamique
        var displayTeamName = teamNames.length > 0 ? teamNames.join(' & ') : "Périmètre complet";

        // -------------------------------------------------------------
        // 2. TRI PAR POSTE DE TRAVAIL PUIS PAR NOM
        // -------------------------------------------------------------
        combinedMembers.sort(function(a, b) {
            var posteA = (a.posteDeTravail || "Z_Non assigné").toUpperCase();
            var posteB = (b.posteDeTravail || "Z_Non assigné").toUpperCase();
            
            if (posteA < posteB) return -1;
            if (posteA > posteB) return 1;
            
            var nomA = (a.nom || "").toUpperCase();
            var nomB = (b.nom || "").toUpperCase();
            
            if (nomA < nomB) return -1;
            if (nomA > nomB) return 1;
            
            return 0;
        });
